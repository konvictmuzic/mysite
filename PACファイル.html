function FindProxyForURL(url, host) {

    var ip = dnsResolve(host); // ホストのIPアドレスを取得

    if (
        LocalHostOrDomainIs(host, "localhost") ||
        isInNet(ip, "10.0.0.0", "255.0.0.0") ||
        isInNet(ip, "127.0.0.0", "255.0.0.0") ||
        isInNet(ip, "172.16.0.0", "255.240.0.0") ||
        isInNet(ip, "192.168.0.0", "255.255.0.0") ||
        dnsDomainIs(host, "a-sha.jp")
    ) {
        return "DIRECT"; // ローカル接続は直接アクセス
    }

    if (
        dnsDomainIs(host, "image.cdn.example") ||
        shExpMatch(host, "*.c-saas.example")
    ) {
        return "PROXY proxy.osaka.a-sha.jp:8080"; // 特定ホストは大阪のプロキシを使用
    }

    return "PROXY proxy.a-sha.jp:8080"; // それ以外は通常プロキシ
}

コードの説明

FindProxyForURL(url, host) は、アクセスするURLのホスト名をもとにプロキシ設定を決定する関数です。

  • dnsResolve(host) - ホスト名をIPアドレスに変換
  • LocalHostOrDomainIs(host, "localhost") - ホスト名が "localhost" の場合は直接接続
  • isInNet(ip, "10.0.0.0", "255.0.0.0") - IPがプライベートアドレスの場合は直接接続
  • dnsDomainIs(host, "image.cdn.example") - 特定ホストはプロキシ `proxy.osaka.a-sha.jp` を使用
  • 上記以外のホストは通常のプロキシ `proxy.a-sha.jp` を使用
